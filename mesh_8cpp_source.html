113 void Mesh<int, double>::
114 generateMesh(int N, double xmin, double xmax)
115 {
116  // Check to make sure a valid mesh is created.
117  // Check that N is positive so mesh arrays can be allocated.
118  if (N <= 0)
119  throw std::invalid_argument(
120  "N is not positive in Mesh<int, double>::generateMesh");
121  /* Check that xmin < xmax so coordinates are monotonically
122  * increasing. */
123  if (xmin >= xmax)
124  throw std::invalid_argument(
125  "xmin >= xmax in Mesh<int, double>::generateMesh" );
126  // Set number of meshpoints.
127  num_meshpoints = N;
128  // Set number of cells.
129  num_cells = N - 1;
130  // Only one indexing dimension.
131  num_indices = num_meshpoints;
132  // Resize (allocate) vector arrays.
133  coordinates.resize(num_meshpoints);
134  laplace.resize(num_meshpoints, num_meshpoints);
135  // Calculate cell size.
136  // Cells are only evenly sized at the moment.
137  double dx = (xmax - xmin)/(N - 1);
138  // Set the index and coordinate values.
139  for (int i=0; i < num_meshpoints; i++) {
140  coordinates[i] = xmin + dx*i;
141  }
142 }

180 template <>
181 double Mesh<int, double>::getWeight(double x, int vert) const
182 {
183  // Get distance from vertex and test which side the particle is on.
184  /* While dx is constant for an evenly spaced mesh it is not saved.
185  * This is because an evenly spaced mesh is not expected to be
186  * permanent. This calculation also assumes the particle size is
187  * equal to the mesh spacing. */
188  double xi_diff = std::abs(x - coordinates[vert])
189  /(coordinates[vert] - coordinates[vert-1]);
190  if (xi_diff < 0.5) {
191  return 0.75 - xi_diff*xi_diff;
192  } else if( xi_diff > 1.5) {
193  return 0.0;
194  } else {
195  return 0.5*xi_diff*xi_diff - 1.5*xi_diff + 1.125;
196  }
197 }

208 {
209  // Assume all element sizes are equal.
210  double dx = coordinates[1] - coordinates[0];
211  /* Eigen claims that this method is efficient for populating the
212  * matrix. */
213  laplace.reserve(Eigen::VectorXi::Constant(num_meshpoints, 3));
214  for(int i = 0; i < num_meshpoints; i++) {
215  if(i == 0) {
216  laplace.insert(i, i) = 2.0/dx;
217  laplace.insert(i, i+1) = -1.0/dx;
218  } else if(i == num_meshpoints - 1) {
219  laplace.insert(i, i-1) = -1.0/dx;
220  laplace.insert(i, i) = 2.0/dx;
221  } else {
222  laplace.insert(i, i-1) = -1.0/dx;
223  laplace.insert(i, i) = 2.0/dx;
224  laplace.insert(i, i+1) = -1.0/dx;
